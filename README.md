    private void jButton1ActionPerformed(java.awt.event.ActionEvent evt) {//GEN-FIRST:event_jButton1ActionPerformed
       state = true;
       
       Thread t = new Thread(){
           public void run()
           {
                for(;;)
               {
                   if (state == true)
                    {
                        try
                        {
                            sleep(1);
                            
                            if (milliseconds>1000)
                            {
                                milliseconds = 0;
                                seconds++;
                            }
                            if (seconds>60)
                            {
                                milliseconds = 0;
                                seconds = 0;
                                minutes++;
                            }
                            if (minutes>60)
                            {
                                milliseconds = 0;
                                seconds = 0;
                                minutes = 0;
                                hours++;
                            }
                            Millisecond.setText(" : " + milliseconds);
                             Second.setText(" : " + seconds);
                              Minute.setText(" : " + minutes);
                               Hour.setText("" + hours);
                            
                            milliseconds++;
                        }
                        catch (Exception e)
                        {
                            
                        }
                    }
                   else
                   {
                       break;
                   }
               }
               
         
              
           }
       };
       t.start();
    }//GEN-LAST:event_jButton1ActionPerformed
