    private void SATstartActionPerformed(java.awt.event.ActionEvent evt) {                                         
        time = new Timer(1000, new ActionListener() {   //main code, makes the timer run

            @Override
            public void actionPerformed(ActionEvent e) {                
                if(stop){
                    min = min1;
                    sec = sec1;
                    stop = false;
                }                
                if (sec == 0) {
                    sec = 60;
                    min--;
                }                
                if (min < 0) {
                    JOptionPane.showMessageDialog(rootPane, "Times up! ", "Stopped", 0);
                    min = 0;
                    sec = 0;
                } else {
                    sec--;
                    if (sec < 10) {
                        SATsec.setText("0" + sec);
                        flag = false;
                    }
                    if (min < 10) {
                        SATmin.setText("0" + min);
                        if (sec < 10) {
                            SATsec.setText("0" + sec);
                        } else {
                            SATsec.setText("" + sec);
                        }
                        flag = false;
                    }
                    if (flag) {
                        SATmin.setText("" + min);
                        SATsec.setText("" + sec);
                    }
                }
            }
        });
        time.start();
    }                                        

    private void SATstopActionPerformed(java.awt.event.ActionEvent evt) {                                        
        min1 = min;
        sec1 = sec;
        stop = true;
        time.stop();
    }
